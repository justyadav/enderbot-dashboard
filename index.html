async function fetchBotStats() { try { const response = await fetch('https://enderbot-backend.onrender.com/api/status'); const data = await response.json(); if (data.status === 'online') { document.getElementById('serverCount').textContent = data.stats.guilds + '+'; document.getElementById('userCount').textContent = data.stats.users.toLocaleString() + '+'; } } catch (error) { console.log('Could not fetch live stats.'); } } document.addEventListener('DOMContentLoaded', fetchBotStats);
